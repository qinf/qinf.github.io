# 查看磁盘分区
sudo fdisk -l
# 挂载磁盘
sudo mount /dev/sdb /data/
# 查看磁盘挂载UUID
sudo blkid /dev/sdb
# 修改/etc/fstab文件
sudo vi /etc/fstab
# 添加磁盘信息，[UUID=************] [挂载磁盘分区]  [挂载磁盘格式]  0  2
# 第一数字0，0是开机不检查磁盘，1是开机检查磁盘
# 第二个数2，0表示交换分区，1表示启动分区，2表示普通分区 
UUID=${UUID} /media/qinf/D ntfs defaults  0  2
